65public:
69 Status(NeighborList<Index_, Float_> neighbors, Options options) :

71 my_dY(my_neighbors.size() * num_dim_),

73 my_gains(my_neighbors.size() * num_dim_, 1.0),

146 void run(Float_* Y, int limit) {
147 Float_ multiplier = (my_iter < my_options.stop_lying_iter ? my_options.exaggeration_factor : 1);
148 Float_ momentum = (my_iter < my_options.mom_switch_iter ? my_options.start_momentum : my_options.final_momentum);
149
150 for(; my_iter < limit; ++my_iter) {
151 // Stop lying about the P-values after a while, and switch momentum
152 if (my_iter == my_options.stop_lying_iter) {
153 multiplier = 1;
154 }
155 if (my_iter == my_options.mom_switch_iter) {
156 momentum = my_options.final_momentum;
157 }
158
159 iterate(Y, multiplier, momentum);
160 }
161 }

177 static Float_ sign(Float_ x) {
178 constexpr Float_ zero = 0;
179 constexpr Float_ one = 1;
180 return (x == zero ? zero : (x < zero ? -one : one));
181 }

183 void iterate(Float_* Y, Float_ multiplier, Float_ momentum) {
184 compute_gradient(Y, multiplier);
185
186 // Update gains
187 size_t ngains = my_gains.size();
188#ifdef _OPENMP
189 #pragma omp simd
190#endif
191 for (size_t i = 0; i < ngains; ++i) {
192 Float_& g = my_gains[i];
193 constexpr Float_ lower_bound = 0.01;
194 constexpr Float_ to_add = 0.2;
195 constexpr Float_ to_mult = 0.8;
196 g = std::max(lower_bound, sign(my_dY[i]) != sign(my_uY[i]) ? (g + to_add) : (g * to_mult));
197 }
198
199 // Perform gradient update (with momentum and gains)
200#ifdef _OPENMP
201 #pragma omp simd
202#endif
203 for (size_t i = 0; i < ngains; ++i) {
204 my_uY[i] = momentum * my_uY[i] - my_options.eta * my_gains[i] * my_dY[i];
205 Y[i] += my_uY[i];
206 }
207
208 // Make solution zero-mean
209 size_t N = num_observations();
210 for (int d = 0; d < num_dim_; ++d) {
211 auto start = Y + d;
212
213 // Compute means from column-major coordinates.
214 Float_ sum = 0;
215 for (size_t i = 0; i < N; ++i, start += num_dim_) {
216 sum += *start;
217 }
218 sum /= N;
219
220 start = Y + d;
221 for (size_t i = 0; i < N; ++i, start += num_dim_) {
222 *start -= sum;
223 }
224 }
225
226 return;
227 }
228
229private:
230 void compute_gradient(const Float_* Y, Float_ multiplier) {
231 my_tree.set(Y);
232 compute_edge_forces(Y, multiplier);
233
234 size_t N = num_observations();
235 std::fill(my_neg_f.begin(), my_neg_f.end(), 0);
236
237 Float_ sum_Q = compute_non_edge_forces();
238
239 // Compute final t-SNE gradient
240 size_t ntotal = N * static_cast<size_t>(num_dim_);
241 for (size_t i = 0; i < ntotal; ++i) {
242 my_dY[i] = my_pos_f[i] - (my_neg_f[i] / sum_Q);
243 }
244 }

246 void compute_edge_forces(const Float_* Y, Float_ multiplier) {
247 std::fill(my_pos_f.begin(), my_pos_f.end(), 0);
248 size_t N = num_observations();
249
250#ifndef QDTSNE_CUSTOM_PARALLEL
251#ifdef _OPENMP
252 #pragma omp parallel num_threads(my_options.num_threads)
253#endif
254 {
255#ifdef _OPENMP
256 #pragma omp for
257#endif
258 for (size_t n = 0; n < N; ++n) {
259#else
260 QDTSNE_CUSTOM_PARALLEL(N, [&](size_t first_, size_t last_) -> void {
261 for (size_t n = first_; n < last_; ++n) {
262#endif
263
264 const auto& current = my_neighbors[n];
265 size_t offset = n * static_cast<size_t>(num_dim_); // cast to avoid overflow.
266 const Float_* self = Y + offset;
267 Float_* pos_out = my_pos_f.data() + offset;
268
269 for (const auto& x : current) {
270 Float_ sqdist = 0;
271 const Float_* neighbor = Y + static_cast<size_t>(x.first) * num_dim_; // cast to avoid overflow.
272 for (int d = 0; d < num_dim_; ++d) {
273 Float_ delta = self[d] - neighbor[d];
274 sqdist += delta * delta;
275 }
276
277 const Float_ mult = multiplier * x.second / (static_cast<Float_>(1) + sqdist);
278 for (int d = 0; d < num_dim_; ++d) {
279 pos_out[d] += mult * (self[d] - neighbor[d]);
280 }
281 }
282
283#ifndef QDTSNE_CUSTOM_PARALLEL
284 }
285 }
286#else
287 }
288 }, my_options.num_threads);
289#endif
290
291 return;
292 }

294 Float_ compute_non_edge_forces() {
295 size_t N = num_observations();
296
297 if (my_options.leaf_approximation) {
298 my_tree.compute_non_edge_forces_for_leaves(my_options.theta, my_leaf_workspace, my_options.num_threads);
299 }
300
301#if defined(_OPENMP) || defined(QDTSNE_CUSTOM_PARALLEL)
302 if (my_options.num_threads > 1) {
303 // Don't use reduction methods, otherwise we get numeric imprecision
304 // issues (and stochastic results) based on the order of summation.
305
306#ifndef QDTSNE_CUSTOM_PARALLEL
307#ifdef _OPENMP
308 #pragma omp parallel num_threads(my_options.num_threads)
309#endif
310 {
311#ifdef _OPENMP
312 #pragma omp for
313#endif
314 for (size_t n = 0; n < N; ++n) {
315#else
316 QDTSNE_CUSTOM_PARALLEL(N, [&](size_t first_, size_t last_) -> void {
317 for (size_t n = first_; n < last_; ++n) {
318#endif
319
320 auto neg_ptr = my_neg_f.data() + n * static_cast<size_t>(num_dim_); // cast to avoid overflow.
321 if (my_options.leaf_approximation) {
322 my_parallel_buffer[n] = my_tree.compute_non_edge_forces_from_leaves(n, neg_ptr, my_leaf_workspace);
323 } else {
324 my_parallel_buffer[n] = my_tree.compute_non_edge_forces(n, my_options.theta, neg_ptr);
325 }
326
327#ifndef QDTSNE_CUSTOM_PARALLEL
328 }
329 }
330#else
331 }
332 }, my_options.num_threads);
333#endif
334
335 return std::accumulate(my_parallel_buffer.begin(), my_parallel_buffer.end(), static_cast<Float_>(0));
336 }
337#endif
338
339 Float_ sum_Q = 0;
340 for (size_t n = 0; n < N; ++n) {
341 auto neg_ptr = my_neg_f.data() + n * static_cast<size_t>(num_dim_); // cast to avoid overflow.
342 if (my_options.leaf_approximation) {
343 sum_Q += my_tree.compute_non_edge_forces_from_leaves(n, neg_ptr, my_leaf_workspace);
344 } else {
345 sum_Q += my_tree.compute_non_edge_forces(n, my_options.theta, neg_ptr);
346 }
347 }
348 return sum_Q;
349 }
