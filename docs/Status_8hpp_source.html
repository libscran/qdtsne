65class Status {
66public:
70 Status(NeighborList<Index_, Float_> neighbors, Options options) :
71 my_neighbors(std::move(neighbors)),
72 my_tree(my_neighbors.size(), options.max_depth),
73 my_options(std::move(options))
74 {
75 std::size_t full_size = static_cast<std::size_t>(my_neighbors.size()) * num_dim_; // cast to avoid overflow.
76 my_dY.resize(full_size);
77 my_uY.resize(full_size);
78 my_gains.resize(full_size, static_cast<Float_>(1));
79 my_pos_f.resize(full_size);
80 my_neg_f.resize(full_size);

83 my_parallel_buffer.resize(my_neighbors.size());
84 }

149 void run(Float_* Y, int limit) {
150 Float_ multiplier = (my_iter < my_options.stop_lying_iter ? my_options.exaggeration_factor : 1);
151 Float_ momentum = (my_iter < my_options.mom_switch_iter ? my_options.start_momentum : my_options.final_momentum);
152
153 for(; my_iter < limit; ++my_iter) {
154 // Stop lying about the P-values after a while, and switch momentum
155 if (my_iter == my_options.stop_lying_iter) {
156 multiplier = 1;
157 }
158 if (my_iter == my_options.mom_switch_iter) {
159 momentum = my_options.final_momentum;
160 }
161
162 iterate(Y, multiplier, momentum);
163 }
164 }

175 void run(Float_* Y) {
176 run(Y, my_options.max_iterations);

179private:
180 static Float_ sign(Float_ x) {
181 constexpr Float_ zero = 0;
182 constexpr Float_ one = 1;
183 return (x == zero ? zero : (x < zero ? -one : one));
184 }
185
186 void iterate(Float_* Y, Float_ multiplier, Float_ momentum) {
187 compute_gradient(Y, multiplier);
188
189 // Update gains
190 auto ngains = my_gains.size();
191 for (decltype(ngains) i = 0; i < ngains; ++i) {
192 Float_& g = my_gains[i];
193 constexpr Float_ lower_bound = 0.01;
194 constexpr Float_ to_add = 0.2;
195 constexpr Float_ to_mult = 0.8;
196 g = std::max(lower_bound, sign(my_dY[i]) != sign(my_uY[i]) ? (g + to_add) : (g * to_mult));
197 }
198
199 // Perform gradient update (with momentum and gains)
200 for (decltype(ngains) i = 0; i < ngains; ++i) {
201 my_uY[i] = momentum * my_uY[i] - my_options.eta * my_gains[i] * my_dY[i];
202 Y[i] += my_uY[i];
203 }
204
205 // Make solution zero-mean for each dimension
206 std::size_t num_obs = num_observations();
207 for (std::size_t d = 0; d < num_dim_; ++d) {
208 Float_ sum = 0;
209 for (std::size_t i = 0; i < num_obs; ++i) {
210 sum += Y[d + i * num_dim_]; // all of these are already size_t to avoid overflow.
211 }
212 sum /= num_obs;
213 for (std::size_t i = 0; i < num_obs; ++i) {
214 Y[d + i * num_dim_] -= sum; // again, everything here is already a size_t.
215 }
216 }
217
218 return;
219 }
220
221private:
222 void compute_gradient(const Float_* Y, Float_ multiplier) {
223 my_tree.set(Y);
224 compute_edge_forces(Y, multiplier);
225
226 std::size_t num_obs = num_observations();
227 std::fill(my_neg_f.begin(), my_neg_f.end(), 0);
228 Float_ sum_Q = compute_non_edge_forces();
229
230 // Compute final t-SNE gradient
231 std::size_t ntotal = num_obs * num_dim_; // already size_t's to avoid overflow.
232 for (std::size_t i = 0; i < ntotal; ++i) {
233 my_dY[i] = my_pos_f[i] - (my_neg_f[i] / sum_Q);
234 }
235 }
236
237 void compute_edge_forces(const Float_* Y, Float_ multiplier) {
238 std::fill(my_pos_f.begin(), my_pos_f.end(), 0);
239 std::size_t num_obs = num_observations();
240
241 parallelize(my_options.num_threads, num_obs, [&](int, std::size_t start, std::size_t length) -> void {
242 for (std::size_t i = start, end = start + length; i < end; ++i) {
243 const auto& current = my_neighbors[i];
244 size_t offset = i * num_dim_; // already size_t's to avoid overflow.
245 const Float_* self = Y + offset;
246 Float_* pos_out = my_pos_f.data() + offset;
247
248 for (const auto& x : current) {
249 Float_ sqdist = 0;
250 const Float_* neighbor = Y + static_cast<std::size_t>(x.first) * num_dim_; // cast to avoid overflow.
251 for (std::size_t d = 0; d < num_dim_; ++d) {
252 Float_ delta = self[d] - neighbor[d];
253 sqdist += delta * delta;
254 }
255
256 const Float_ mult = multiplier * x.second / (static_cast<Float_>(1) + sqdist);
257 for (std::size_t d = 0; d < num_dim_; ++d) {
258 pos_out[d] += mult * (self[d] - neighbor[d]);
259 }
260 }
261 }
262 });
263
264 return;
265 }
266
267 Float_ compute_non_edge_forces() {
268 if (my_options.leaf_approximation) {
269 my_tree.compute_non_edge_forces_for_leaves(my_options.theta, my_leaf_workspace, my_options.num_threads);
270 }
271
272 std::size_t num_obs = num_observations();
273 if (my_options.num_threads > 1) {
274 // Don't use reduction methods, otherwise we get numeric imprecision
275 // issues (and stochastic results) based on the order of summation.
276 parallelize(my_options.num_threads, num_obs, [&](int, std::size_t start, std::size_t length) -> void {
277 for (std::size_t i = start, end = start + length; i < end; ++i) {
278 auto neg_ptr = my_neg_f.data() + i * num_dim_; // already size_t's to avoid overflow.
279 if (my_options.leaf_approximation) {
280 my_parallel_buffer[n] = my_tree.compute_non_edge_forces_from_leaves(n, neg_ptr, my_leaf_workspace);
281 } else {
282 my_parallel_buffer[n] = my_tree.compute_non_edge_forces(n, my_options.theta, neg_ptr);
283 }
284 }
285 });
286 return std::accumulate(my_parallel_buffer.begin(), my_parallel_buffer.end(), static_cast<Float_>(0));
287 }
288
289 Float_ sum_Q = 0;
290 for (std::size_t i = 0; i < num_obs; ++i) {
291 auto neg_ptr = my_neg_f.data() + i * num_dim_; // already size_ts to avoid overflow.
292 if (my_options.leaf_approximation) {
293 sum_Q += my_tree.compute_non_edge_forces_from_leaves(n, neg_ptr, my_leaf_workspace);
294 } else {
295 sum_Q += my_tree.compute_non_edge_forces(n, my_options.theta, neg_ptr);
296 }
297 }
298 return sum_Q;
299 }
