85template<int num_dim_, typename Index_, typename Float_>
86Status<num_dim_, Index_, Float_> initialize(NeighborList<Index_, Float_> neighbors, const Options& options) {
87 Float_ perp;
88 if (options.infer_perplexity && neighbors.size()) {
89 perp = static_cast<Float_>(neighbors.front().size())/3;
90 } else {
91 perp = options.perplexity;
92 }
93 return internal::initialize<num_dim_>(std::move(neighbors), perp, options);
94}

110template<int num_dim_, typename Dim_, typename Index_, typename Float_>
111Status<num_dim_, Index_, Float_> initialize(const knncolle::Prebuilt<Dim_, Index_, Float_>& prebuilt, const Options& options) {
112 // TODO: replace max_neighbors with knncolle::cap_nearest_neighbors().
113 const Index_ max_neighbors = std::max(static_cast<Index_>(1), prebuilt.num_observations()) - 1;
114 const Index_ K = std::min(perplexity_to_k(options.perplexity), max_neighbors);
115 auto neighbors = find_nearest_neighbors(prebuilt, K, options.num_threads);
116 return internal::initialize<num_dim_>(std::move(neighbors), options.perplexity, options);
117}
