66template<std::size_t num_dim_, typename Float_ = double>
67void initialize_random(Float_* const Y, const std::size_t num_points, const unsigned long long seed = 42) {
68 // The constructor accepts an unsigned type, so any overflow should just wrap around harmlessly.
69 std::mt19937_64 rng(seed);
70
71 // Presumably a size_t can store the product in order to allocate Y in the first place.
72 std::size_t num_total = sanisizer::product_unsafe<std::size_t>(num_points, num_dim_);
73 const bool odd = num_total % 2;
74 if (odd) {
75 --num_total;
76 }
77
78 // Box-Muller gives us two random values at a time.
79 for (std::size_t i = 0; i < num_total; i += 2) {
80 auto paired = aarand::standard_normal<Float_>(rng);
81 Y[i] = paired.first;
82 Y[i + 1] = paired.second;
83 }
84
85 if (odd) {
86 // Adding the poor extra for odd total lengths.
87 auto paired = aarand::standard_normal<Float_>(rng);
88 Y[num_total] = paired.first;
89 }
90
91 return;
92}
